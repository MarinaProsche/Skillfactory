def win_no_one():
    step = 0
    for key in numbers:
        if numbers[key] == "|  X" or numbers[key] == "|  0":
            step +=1
    if step > 7:
        print("Победила дружба!")
        quit()

@check_cross
def cross():
    a = input('Введите код квадрата для крестика ("a1", "b2"): ')
    if a not in numbers:
        print("Неверное число. Попробуйте еще раз")
        cross()
    else:
        for key in numbers:
            if key == a:
                if numbers[key] == "|   ":
                    numbers[key] = "|  X"
                    return numbers
                else:
                    print("Квадратик уже занят!")
                    cross()
